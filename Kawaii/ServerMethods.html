def start!(port) # @todo Support other handlers http://www.rubydoc.info/github/rack/rack/Rack/Handler
  Rack::Handler.get(WEBRICK).run(self, Port: port) do |s|
    @server = s
    at_exit { stop! }
    [:INT, :TERM].each do |signal|
      old = trap(signal) do
        stop!
        old.call if old.respond_to?(:call)
      end
    end
  end
end
